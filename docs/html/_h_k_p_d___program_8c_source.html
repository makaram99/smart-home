61 u8 HKPD_U8GetKeyPressed(void)
62 {
63  u8 LOC_U8RowCount;
64  u8 LOC_U8ColCount;
65  /*Initialize the switch status to NOT PRESSED*/
66  u8 LOC_U8ReturnValue=NOT_PRESSED;
67  u8 LOC_U8PinState;
68  /*Looping on columns of the keypad*/
69  for(LOC_U8ColCount=COL_INIT;LOC_U8ColCount<=COL_END;LOC_U8ColCount++)
70  {
71  /*Active the Column */
72  MDIO_Error_State_SetPinValue(LOC_U8ColCount,COL_PORT,PIN_LOW);
73  /*Loop to read the all row pins*/
74  for(LOC_U8RowCount=ROW_INIT;LOC_U8RowCount<=ROW_END;LOC_U8RowCount++)
75  {
76  /*check the status of the switch*/
77  MDIO_Error_State_GetPinValue(LOC_U8RowCount,ROW_PORT,&LOC_U8PinState);
78  if(LOC_U8PinState==0)
79  {
80  /*Get the Value of the current pressed switch*/
81  LOC_U8ReturnValue=KPD_u8SwitchVal[LOC_U8ColCount-COL_INIT][LOC_U8RowCount-ROW_INIT];
82  /*wait until the switch is released(Single Press)*/
83  while(LOC_U8PinState==0)
84  {
85  MDIO_Error_State_GetPinValue(LOC_U8RowCount,ROW_PORT,&LOC_U8PinState);
86  }
87  /*Delay To avoid Bouncing*/
88  _delay_ms(10);
89  }
90  else
91  {
92  /*Do Nothing*/
93  }
94  }
95  /*Deactivate the Column*/
96  MDIO_Error_State_SetPinValue(LOC_U8ColCount,COL_PORT,PIN_HIGH);
97  }
98  return LOC_U8ReturnValue;
